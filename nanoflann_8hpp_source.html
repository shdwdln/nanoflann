00467         class PooledAllocator
00468         {
00469                 /* We maintain memory alignment to word boundaries by requiring that all
00470                     allocations be in multiples of the machine wordsize.  */
00471                 /* Size of machine word in bytes.  Must be power of 2. */
00472                 /* Minimum number of bytes requested at a time from     the system.  Must be multiple of WORDSIZE. */
00473 
00474 
00475                 size_t  remaining;  /* Number of bytes left in current block of storage. */
00476                 void*   base;     /* Pointer to base of current block of storage. */
00477                 void*   loc;      /* Current location in block to next allocate memory. */
00478 
00479                 void internal_init()
00480                 {
00481                         remaining = 0;
00482                         base = NULL;
00483                         usedMemory = 0;
00484                         wastedMemory = 0;
00485                 }
00486 
00487         public:
00488                 size_t  usedMemory;
00489                 size_t  wastedMemory;
00490 
00494                 PooledAllocator() {
00495                         internal_init();
00496                 }
00497 
00501                 ~PooledAllocator() {
00502                         free_all();
00503                 }
00504 
00506                 void free_all()
00507                 {
00508                         while (base != NULL) {
00509                                 void *prev = *(static_cast<void**>( base)); /* Get pointer to prev block. */
00510                                 ::free(base);
00511                                 base = prev;
00512                         }
00513                         internal_init();
00514                 }
00515 
00520                 void* malloc(const size_t req_size)
00521                 {
00522                         /* Round size up to a multiple of wordsize.  The following expression
00523                             only works for WORDSIZE that is a power of 2, by masking last bits of
00524                             incremented size to zero.
00525                          */
00526                         const size_t size = (req_size + (WORDSIZE - 1)) & ~(WORDSIZE - 1);
00527 
00528                         /* Check whether a new block must be allocated.  Note that the first word
00529                             of a block is reserved for a pointer to the previous block.
00530                          */
00531                         if (size > remaining) {
00532 
00533                                 wastedMemory += remaining;
00534 
00535                                 /* Allocate new storage. */
00536                                 const size_t blocksize = (size + sizeof(void*) + (WORDSIZE-1) > BLOCKSIZE) ?
00537                                                         size + sizeof(void*) + (WORDSIZE-1) : BLOCKSIZE;
00538 
00539                                 // use the standard C malloc to allocate memory
00540                                 void* m = ::malloc(blocksize);
00541                                 if (!m) {
00542                                         fprintf(stderr,"Failed to allocate memory.\n");
00543                                         return NULL;
00544                                 }
00545 
00546                                 /* Fill first word of new block with pointer to previous block. */
00547                                 static_cast<void**>(m)[0] = base;
00548                                 base = m;
00549 
00550                                 size_t shift = 0;
00551                                 //int size_t = (WORDSIZE - ( (((size_t)m) + sizeof(void*)) & (WORDSIZE-1))) & (WORDSIZE-1);
00552 
00553                                 remaining = blocksize - sizeof(void*) - shift;
00554                                 loc = (static_cast<char*>(m) + sizeof(void*) + shift);
00555                         }
00556                         void* rloc = loc;
00557                         loc = static_cast<char*>(loc) + size;
00558                         remaining -= size;
00559 
00560                         usedMemory += size;
00561 
00562                         return rloc;
00563                 }

01214                                                  distance_vector_t& dists, const float epsError) const

01251                         }
